51  virtual btScalar addSingleResult(btCollisionWorld::LocalRayResult &rayResult, bool normalInWorldSpace)
52  {
53  //caller already does the filter on the m_closestHitFraction
54  btAssert(rayResult.m_hitFraction <= m_closestHitFraction);
55 
56  m_closestHitFraction = rayResult.m_hitFraction;
57  m_collisionObject = rayResult.m_collisionObject;
58 
59  btVector3 normal;
60  if(normalInWorldSpace)
61  {
62  normal = rayResult.m_hitNormalLocal;
63  }
64  else
65  {
66  normal = m_collisionObject->getWorldTransform().getBasis()*rayResult.m_hitNormalLocal;
67  }
68 
69  // m_hitPointWorld.setInterpolate3(m_rayFromWorld, m_rayToWorld, rayResult.m_hitFraction);
70  result.hasHit = true;
71  result.hitFractions.push() = rayResult.m_hitFraction;
72  result.hitNormalWorlds.push() = CONST_CAST_VECTOR3(normal);
73  result.hitPointWorlds.push() = result.rayFromWorld * (1.0f-rayResult.m_hitFraction) +
74  result.rayToWorld * rayResult.m_hitFraction;
75  result.hitCollisionObjects.push() = rayResult.m_collisionObject->getUserPointer();
76 
77  return rayResult.m_hitFraction;
78  }
